Dependencies:

# 1. Upgrade pip to the latest version (26.0.1+)
python -m pip install --upgrade pip

# 2. Clean out the old, conflicting database drivers
pip uninstall -y pymongo motor beanie

# 3. Install Web Server & Framework
pip install fastapi "uvicorn[standard]"

# 4. Install Database (Beanie pulls in the correct Motor/PyMongo automatically)
pip install beanie pydantic-settings python-dotenv

# 5. Install Security & Authentication
pip install "python-jose[cryptography]" "passlib[bcrypt]" pyjwt authlib httpx

# 6. Install Validation & Email
pip install "pydantic[email]" fastapi-mail

pip install -U motor
